1.  a_rattuppaal
1.1        paayiraviyal
1.1.1      ka.tavu.l vaa_lttu
1.         akara mutala e_luttellaam aati
           pakava_n muta_r_ree ulaku.
2.         ka_r_rata_naal aaya paya_ne_nkol vaala_riva_n
           na_r_raa_n to_laaar e_ni_n.
3.         malarmicai eeki_naa_n maa.na.ti ceerntaar
           nilamicai nii.tuvaa_l vaar.
4.         vee.n.tutal vee.n.taamai ilaa_na.ti ceerntaarkku
           yaa.n.tum i.tumpai ila.
5.         iru.lceer iruvi_naiyum ceeraa i_raiva_n
           poru.lceer puka_lpurintaar maa.t.tu.
6.         po_rivaayil aintavittaa_n poytiir o_lukka
           ne_rini_n_raar nii.tuvaa_l vaar.
7.         ta_nakkuvamai illaataa_n taa.lceerntaark kallaal
           ma_nakkavalai maa_r_ral aritu.
8.         a_ravaa_li anta.na_n taa.lceerntaark kallaal
           pi_ravaa_li niintal aritu.
9.         koo.lil po_riyi_r ku.namilavee e.nku.nattaa_n
           taa.lai va.na^nkaat talai.
10.        pi_ravip peru^nka.tal niintuvar niintaar
           i_raiva_n a.ticeeraa taar.

1.1.2      vaa_nci_rappu
11.        vaa_ni_n_ru ulakam va_la^nki varutalaal
           taa_nami_ltam e_n_ru.nara_r paa_r_ru.
12.        tuppaarkkut tuppaaya tuppaakkit tuppaarkkut
           tuppaaya tuuum ma_lai.
13.        vi.ni_n_ru poyppi_n viriniir viya_nulakattu
           u.lni_n_ru u.ta_r_rum paci.
14.        eeri_n u_laaar u_lavar puyale_n_num
           vaari va.la^nku_n_rik kaal.
15.        ke.tuppatuuum ke.t.taarkkuc caarvaayma_r _raa^nkee
           e.tuppatuuum ellaam ma_lai.
16.        vicumpi_n tu.livii_li_n allaalma_r _raa^nkee
           pacumpul talaikaa.npu aritu.
17.        ne.tu^nka.talum ta_nniirmai ku_n_rum ta.tinte_lili
           taa_nnalkaa taaki vi.ti_n.
18.        ci_rappo.tu puuca_nai cellaatu vaa_nam
           va_rakkumeel vaa_noorkkum ii.n.tu.
19.        taa_nam tavamira.n.tum ta^nkaa viya_nulakam
           vaa_nam va_la^nkaa te_ni_n.
20.        niiri_n_ru amaiyaatu ulake_ni_n yaaryaarkkum
           vaa_ni_n_ru amaiyaatu o_lukku.

           vee.n.tum pa_nuval tu.nivu.

           i_rantaarai e.n.nikko.n .ta_r_ru.
23.        irumai vakaiterintu ii.n.tua_ram puu.n.taar

24.        ura_ne_n_nun too.t.tiyaa_n ooraintum kaappaa_n
           vara_ne_n_num vaippi_rkoor vittu.
25.        aintavittaa_n aa_r_ral akalvicumpu .laarkoomaa_n
           intira_nee caalu^n kari.
26.        ceya_rkariya ceyvaar periyar ci_riyar
           ceya_rkariya ceykalaa taar.
27.        cuvaiyo.li uu_roocai naa_r_rame_n_ru ainti_n
           vakaiteri vaa_n ka.t.tee ulaku.
28.        ni_raimo_li maantar perumai nilattu
           ma_raimo_li kaa.t.ti vi.tum.
29.        ku.name_n_num ku_n_ree_ri ni_n_raar veku.li
           ka.nameeyu^n kaattal aritu.
30.        anta.nar e_npoor a_ravoorma_r _revvuyirkkum
           centa.nmai puu.n.to_luka laa_n.

1.1.4  a_ra_n valiyu_ruttal

           po_ruttaa_noo.tu uurntaa_n i.tai.
38.        vii_lnaa.l pa.taaamai na_n_raa_r_ri_n a_ktoruva_n
           vaa_lnaa.l va_liya.taikku^n kal.

           pu_ratta puka_lum ila.

           uya_rpaala toorum pa_li.

           ni_raikaakkum kaappee talai.
